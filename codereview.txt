    <script>
      // Инициализация данных
      let requests = JSON.parse(localStorage.getItem('srv_requests_v2')) || [];

      // Добавление 2 рандомных задач при первом запуске
      if (requests.length === 0) {
          requests = [
              {
                  id: Date.now() - 100000,
                  clientName: "Александр Соколов",
                  phone: "+7 (921) 555-12-34",
                  address: "ул. Пушкина, д. 42 / кв. 15",
                  problemText: "Разбито стекло балконной двери, дует сильный сквозняк.",
                  status: "new",
                  assignedTo: "",
                  createdAt: new Date(Date.now() - 3600000 * 2).toISOString(),
                  updatedAt: new Date(Date.now() - 3600000 * 2).toISOString()
              },
              {
                  id: Date.now() - 50000,
                  clientName: "Елена Кузнецова",
                  phone: "+7 (905) 777-88-99",
                  address: "Ленинский пр., д. 100, блок Б",
                  problemText: "Перестал работать бойлер, выбивает пробки при включении.",
                  status: "assigned",
                  assignedTo: "Иванов",
                  createdAt: new Date(Date.now() - 3600000 * 5).toISOString(),
                  updatedAt: new Date(Date.now() - 3600000 * 4).toISOString()
              }
          ];
          localStorage.setItem('srv_requests_v2', JSON.stringify(requests));
      }

      let currentRole = 'client';
      let currentMaster = '';

      // Слушаем изменения в localStorage от других вкладок
      window.addEventListener('storage', (e) => {
          if (e.key === 'srv_requests_v2') {
              // Обновляем локальные данные
              requests = JSON.parse(e.newValue) || [];

              // Перерендериваем текущее представление
              if (currentRole === 'dispatcher') {
                  renderDispatcherRequests();
              } else if (currentRole === 'master' && currentMaster) {
                  renderMasterRequests();
              }

              showToast('🔄 Данные обновлены из другой вкладки');
          }
      });

      // Серверная логика эмуляции предотвращения Race Condition
      const MockServer = {
          async transitionToWork(requestId) {
              await new Promise(r => setTimeout(r, 800));

              const latestRequests = JSON.parse(localStorage.getItem('srv_requests_v2'));
              const req = latestRequests.find(r => r.id === requestId);

              if (!req) return { status: 404, message: 'Заявка не найдена' };

              if (req.status !== 'assigned') {
                  return {
                      status: 409,
                      message: `[409 Conflict] Заявка уже занята. Текущий статус: "${req.status}". Возможно, другой мастер успел перехватить её.`
                  };
              }

              const idx = latestRequests.findIndex(r => r.id === requestId);
              latestRequests[idx].status = 'in_progress';
              latestRequests[idx].updatedAt = new Date().toISOString();

              localStorage.setItem('srv_requests_v2', JSON.stringify(latestRequests));
              requests = JSON.parse(localStorage.getItem('srv_requests_v2'));

              return { status: 200, message: 'Статус обновлен: В работе' };
          }
      };

      function saveData() {
          localStorage.setItem('srv_requests_v2', JSON.stringify(requests));
      }

      function showToast(message) {
          const container = document.getElementById('toastContainer');
          const toast = document.createElement('div');
          toast.className = 'toast';
          toast.innerText = message;
          container.appendChild(toast);
          setTimeout(() => {
              toast.style.opacity = '0';
              toast.style.transform = 'translateY(20px)';
              setTimeout(() => toast.remove(), 300);
          }, 4000);
      }

      function setRole(role, master = '') {
          currentRole = role;
          currentMaster = master;

          document.querySelectorAll('.view').forEach(v => v.classList.add('hidden'));
          document.querySelectorAll('#roleSwitcher button').forEach(b => b.classList.remove('active'));

          if (role === 'client') {
              document.getElementById('view-client').classList.remove('hidden');
              document.getElementById('btn-client').classList.add('active');
          } else if (role === 'dispatcher') {
              document.getElementById('view-dispatcher').classList.remove('hidden');
              document.getElementById('btn-dispatcher').classList.add('active');
              renderDispatcherRequests();
          } else if (role === 'master' && master) {
              document.getElementById('view-master').classList.remove('hidden');
              document.getElementById('currentMasterBadge').innerText = master;
              renderMasterRequests();
          }
      }

      document.getElementById('createRequestForm').onsubmit = (e) => {
          e.preventDefault();
          const newReq = {
              id: Date.now(),
              clientName: document.getElementById('clientName').value,
              phone: document.getElementById('phone').value,
              address: document.getElementById('address').value,
              problemText: document.getElementById('problemText').value,
              status: 'new',
              assignedTo: '',
              createdAt: new Date().toISOString(),
              updatedAt: new Date().toISOString()
          };
          requests.push(newReq);
          saveData();
          showToast('✅ Ваша заявка успешно создана');
          e.target.reset();
      };

      function assignMaster(id) {
          const masterSelector = document.getElementById(`assign-${id}`);
          const master = masterSelector.value;
          if (!master) return;

          const req = requests.find(r => r.id === id);
          req.status = 'assigned';
          req.assignedTo = master;
          req.updatedAt = new Date().toISOString();
          saveData();

          requests = JSON.parse(localStorage.getItem('srv_requests_v2'));

          showToast(`Мастер ${master} назначен`);
          renderDispatcherRequests();
      }

      function cancelRequest(id) {
          const req = requests.find(r => r.id === id);
          req.status = 'canceled';
          req.updatedAt = new Date().toISOString();
          saveData();

          requests = JSON.parse(localStorage.getItem('srv_requests_v2'));

          showToast('Заявка отменена');
          renderDispatcherRequests();
      }

      async function takeInWork(id) {
          const result = await MockServer.transitionToWork(id);
          showToast(result.message);
          renderMasterRequests();
      }

      function completeRequest(id) {
          const req = requests.find(r => r.id === id);
          req.status = 'done';
          req.updatedAt = new Date().toISOString();
          saveData();

          requests = JSON.parse(localStorage.getItem('srv_requests_v2'));

          showToast('Заявка выполнена');
          renderMasterRequests();
      }

      function renderDispatcherRequests() {
          const filter = document.getElementById('filterStatus').value;
          const list = document.getElementById('dispatcherList');
          list.innerHTML = '';

          const filtered = requests.filter(r => filter === 'all' || r.status === filter);

          if (!filtered.length) {
              list.innerHTML = '<div style="text-align:center; padding: 40px; color: var(--text-muted)">Пусто</div>';
              return;
          }

          filtered.sort((a,b) => b.id - a.id).forEach(req => {
              const div = document.createElement('div');
              div.className = 'request-item';
              div.innerHTML = `
                  <div class="request-info">
                      <span class="badge status-${req.status}">${req.status === 'new' ? 'Новая' : req.status === 'assigned' ? 'Ожидает' : req.status === 'in_progress' ? 'В работе' : req.status === 'done' ? 'Завершена' : 'Отменена'}</span>
                      <h4>${req.clientName}</h4>
                      <p class="request-meta">📍 ${req.address} • 📞 ${req.phone}</p>
                      <p class="problem-text">${req.problemText}</p>
                      ${req.assignedTo ? `<div class="info-tag">Ответственный: ${req.assignedTo}</div>` : ''}
                  </div>
                  <div class="actions">
                      ${req.status === 'new' ? `
                          <div class="dispatcher-actions">
                              <select id="assign-${req.id}">
                                  <option value="">Выбор...</option>
                                  <option value="Иванов">Иванов</option>
                                  <option value="Петров">Петров</option>
                              </select>
                              <button onclick="assignMaster(${req.id})" class="primary-btn">OK</button>
                          </div>
                      ` : ''}
                      ${(req.status === 'new' || req.status === 'assigned') ? `<button onclick="cancelRequest(${req.id})" class="btn-cancel">Отменить</button>` : ''}
                  </div>
              `;
              list.appendChild(div);
          });
      }

      function renderMasterRequests() {
          const list = document.getElementById('masterList');
          list.innerHTML = '';

          const myRequests = requests.filter(r => r.assignedTo === currentMaster && r.status !== 'canceled');

          if (!myRequests.length) {
              list.innerHTML = '<div style="text-align:center; padding: 40px; color: var(--text-muted)">У вас пока нет задач</div>';
              return;
          }

          myRequests.sort((a,b) => b.id - a.id).forEach(req => {
              const div = document.createElement('div');
              div.className = 'request-item';
              div.innerHTML = `
                  <div class="request-info">
                      <span class="badge status-${req.status}">${req.status === 'assigned' ? 'Ожидание' : req.status === 'in_progress' ? 'Активна' : 'Выполнена'}</span>
                      <h4>${req.clientName}</h4>
                      <p class="request-meta">🏠 ${req.address}</p>
                      <p class="problem-text">${req.problemText}</p>
                  </div>
                  <div class="actions">
                      ${req.status === 'assigned' ? `<button onclick="takeInWork(${req.id})" class="primary-btn" style="height: 48px">Взять в работу</button>` : ''}
                      ${req.status === 'in_progress' ? `<button onclick="completeRequest(${req.id})" style="background:#22c55e; color:white; border:none; height: 48px">Завершить задачу</button>` : ''}
                  </div>
              `;
              list.appendChild(div);
          });
      }

      // Начальный выбор
      setRole('client');
    </script>
